            function setDateLimits() {
                var currentDate = new Date();
                var formattedDate = currentDate.toISOString().split('T')[0]; // Format: yyyy-mm-dd

                // Set the max date as the current date
                document.getElementById('from_dt').max = formattedDate;
                document.getElementById('ToDate').max = formattedDate;

                // Set the min date as January 1, 2000
                document.getElementById('from_dt').min = '2000-01-01';
                document.getElementById('ToDate').min = '2000-01-01';

            function searching() {
                $('#text_search').on("keyup", function () {
                    var value = $(this).val().toLowerCase();  // Convert search input to lower case
                    $('#tblpen tbody tr').filter(function () {
                        var rowText = $(this).text().toLowerCase();  // Convert row text to lower case
                        $(this).toggle(rowText.indexOf(value) > -1);  // Check if search term is in row text
                    });
                });
            }



            function auth_check() {
                const input = $("[id*=hdUserId]").val();


                // Perform AJAX request to check if the user is E-Head
                $.ajax({
                    type: "POST",
                    contentType: "application/json; charset=utf-8",
                    url: "Lag_Days_Report.aspx/User_auth",
                    data: JSON.stringify({ QueryStr: "Check_user", input: input }),
                    dataType: "json",
                    success: function (response) {
                        const result = response.d;


                        if (result != 1) {
                            alert("Your Are Not Authorised To View This Page");
                            window.location.href = '../Index/index.aspx';
                        }
                        else {
                            $('#form1').show();

                        }
                    },
                    error: function (xhr, status, error) {
                        console.error('Error checking User:', error);
                        alert('An error occurred while checking User.');
                    }
                });
            }
